kubectl get pods -n cert-manager

# if already installed : 
helm uninstall cert-manager -n cert-manager

# Add the Jetstack Helm repository
helm repo add jetstack https://charts.jetstack.io
helm repo update

# Install cert-manager with CRDs
helm install cert-manager jetstack/cert-manager \
    --namespace cert-manager \
    --create-namespace \
    --set installCRDs=true \
    --version v1.16.1 
# Replace with latest stable version

# verification if installed : 
kubectl get crd | grep certificates.cert-manager.io

# If you want to check the specific Certificate resource that's failing:
kubectl get certificate quant-cm-cert -n default
